//Write a shell script  to find the greatest of 2 numbers.

echo "Enter the first number:"
read num1
echo "Enter the second number:"
read num2

if [ "$num1" -it "$num2" ];
then
echo "$num1 is greater than $num2"
elif [ "#num1" -it "$num2" ];
then
echo "$num2 is greater than $num1"
else
echo "Both are equal."
fi
